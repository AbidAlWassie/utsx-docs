## Make your own Custom Component

This is an example of custom dropzone component.

export const DropzoneDemo = () => {
  const handleFileChange = (files) => {
    console.log(files);
  };

  return (
    <Dropzone
      icon={<IoMdImages className="w-12 h-12" />}
      headingText="Upload your images"
      descriptionText="or drag them here"
      subDescriptionText="JPG, PNG, GIF"
      maxSizeText="(Max 5MB)"
      accept="image/jpeg,image/png,image/gif"
      onFileChange={handleFileChange}
    />
  );
};

<DropzoneDemo />
